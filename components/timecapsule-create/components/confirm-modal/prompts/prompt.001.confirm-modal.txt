                - 상태관리: React hooks (useState, useCallback) 사용할 것

            2) 모달 타입 및 사용 시나리오 (총 3가지)

               A) PAYMENT_COMPLETE (결제완료 확인 모달)
                  - 사용 시점: step-info에서 결제 성공 후 자동 표시
                  - 제목: "결제가 완료되었습니다"
                  - 설명: "타임캡슐이 생성되었습니다"
                  - 아이콘: 성공/완료 아이콘 (react-native-remix-icon에서 적절한 아이콘 선택, 예: "ri-checkbox-circle-fill")
                  - 버튼 개수: 1개 (확인)
                  - 확인 버튼 클릭 시: 모달 닫기 및 다음 화면 이동
                  - 뒷배경 클릭: 모달 닫기 가능 (closeOnBackdropPress: true)

               B) SUBMIT_CONFIRM (제출 전 재확인 모달)
                  - 사용 시점: 사용자가 타임캡슐 제출 버튼 클릭 시 표시
                  - 제목: "타임캡슐을 제출하시겠습니까?"
                  - 설명: "제출 후에는 수정할 수 없습니다"
                  - 아이콘: 경고/확인 아이콘 (react-native-remix-icon에서 적절한 아이콘 선택, 예: "ri-alert-fill" 또는 "ri-question-fill")
                  - 버튼 개수: 2개 (취소, 제출)
                  - 버튼 배치:
                    - 좌측: 취소 버튼 (secondary 스타일)
                    - 우측: 제출 버튼 (primary 스타일)
                  - 취소 버튼 클릭 시: 모달만 닫기
                  - 제출 버튼 클릭 시: 제출 로직 실행 및 모달 닫기
                  - 뒷배경 클릭: 모달 닫히지 않음 (closeOnBackdropPress: false) - 실수 방지

               C) SUBMIT_COMPLETE (제출완료 확인 모달)
                  - 사용 시점: 타임캡슐 제출 성공 후 자동 표시
                  - 제목: "제출이 완료되었습니다"
                  - 설명: "타임캡슐이 성공적으로 제출되었습니다"
                  - 아이콘: 성공/완료 아이콘 (react-native-remix-icon에서 적절한 아이콘 선택, 예: "ri-checkbox-circle-fill")
                  - 버튼 개수: 1개 (확인)
                  - 확인 버튼 클릭 시: 모달 닫기 및 이전 화면으로 이동
                  - 뒷배경 클릭: 모달 닫기 가능 (closeOnBackdropPress: true)

            3) 타임캡슐 생성 플로우에서 모달 연결 시나리오
               [step-info 화면]
               ↓ 사용자가 "결제하기" 버튼 클릭
               ↓ 결제 처리 (step-payment 화면으로 이동 또는 결제 API 호출)
               ↓ 결제 성공
               → PAYMENT_COMPLETE 모달 표시
               ↓ 사용자가 "확인" 버튼 클릭
               ↓ 모달 닫기, 다음 단계로 이동

               [타임캡슐 작성 완료 후]
               ↓ 사용자가 "제출하기" 버튼 클릭
               → SUBMIT_CONFIRM 모달 표시 (2개 버튼)
               ↓ 사용자가 "제출" 버튼 클릭
               ↓ 타임캡슐 제출 API 호출
               ↓ 제출 성공
               → SUBMIT_COMPLETE 모달 표시
               ↓ 사용자가 "확인" 버튼 클릭
               ↓ 모달 닫기, 홈 화면 또는 목록 화면으로 이동

   /**
    * 모달 타입 정의
    */
   export type ConfirmModalType = 'PAYMENT_COMPLETE' | 'SUBMIT_CONFIRM' | 'SUBMIT_COMPLETE';

   /**
    * ConfirmModal 컴포넌트 Props
    */
   export interface ConfirmModalProps {
     /** 모달 표시 여부 */
     visible: boolean;
     /** 모달 타입 (3가지 중 선택) */
     type: ConfirmModalType;
     /** 확인 버튼 클릭 시 콜백 */
     onConfirm: () => void;
     /** 취소 버튼 클릭 시 콜백 (SUBMIT_CONFIRM 타입에서만 사용) */
     onCancel?: () => void;
     /** 모달 닫기 콜백 (backdrop 클릭 또는 기타 닫기 이벤트) */
     onClose?: () => void;
   }

   /**
    * 모달 내용 설정 타입
    */
   export interface ModalContentConfig {
     title: string;
     description: string;
     iconName: string;
     iconColor: string;
     confirmText: string;
     cancelText?: string;
     buttonCount: 1 | 2;
     closeOnBackdropPress: boolean;
     width: number; // 각 타입별로 다른 모달 가로 크기
     height: 'auto' | number; // 각 타입별로 다른 모달 세로 크기
   }

3. ConfirmModal 컴포넌트 구현
   - 파일: components/timecapsule-create/components/confirm-modal/index.tsx

   구조:
   ```typescript
   import React, { useCallback, useMemo } from 'react';
   import { View, Text, Pressable } from 'react-native';
   import Modal from '@/commons/components/modal';
   import Icon from 'react-native-remix-icon';
   import { ConfirmModalProps, ConfirmModalType } from './types';
   import { MODAL_CONTENTS, MODAL_DESIGN } from './constants';
   import { styles } from './styles';

   export default function ConfirmModal({
     visible,
     type,
     onConfirm,
     onCancel,
     onClose,
   }: ConfirmModalProps) {
     // type에 따라 constants에서 설정 가져오기
     const config = useMemo(() => MODAL_CONTENTS[type], [type]);

     // 확인 버튼 핸들러
     const handleConfirm = useCallback(() => {
       onConfirm();
     }, [onConfirm]);

     // 취소 버튼 핸들러 (SUBMIT_CONFIRM에서만 사용)
     const handleCancel = useCallback(() => {
       if (onCancel) {
         onCancel();
       }
     }, [onCancel]);

     // 모달 닫기 핸들러
     const handleClose = useCallback(() => {
       if (config.closeOnBackdropPress && onClose) {
         onClose();
       }
     }, [config.closeOnBackdropPress, onClose]);

     return (
       <Modal
         visible={visible}
         onClose={handleClose}
         width={config.width} // 각 타입별로 다른 width 사용
         height={config.height}
         closeOnBackdropPress={config.closeOnBackdropPress}
       >
         <View style={styles.container}>
           {/* 아이콘 영역 */}
           <View style={styles.iconContainer}>
             <Icon
               name={config.iconName}
               size={MODAL_DESIGN.iconSize}
               color={config.iconColor}
             />
           </View>

           {/* 텍스트 영역 */}
           <View style={styles.textContainer}>
             <Text style={styles.title}>{config.title}</Text>
             <Text style={styles.description}>{config.description}</Text>
           </View>

           {/* 버튼 영역 */}
           <View style={styles.buttonContainer}>
             {config.buttonCount === 1 ? (
               // 버튼 1개: 확인만
               <Pressable
                 style={[styles.button, styles.buttonPrimary]}
                 onPress={handleConfirm}
               >
                 <Text style={[styles.buttonText, styles.buttonTextPrimary]}>
                   {config.confirmText}
                 </Text>
               </Pressable>
             ) : (
               // 버튼 2개: 취소, 확인
               <View style={styles.buttonRow}>
                 <Pressable
                   style={[styles.button, styles.buttonSecondary, styles.buttonHalf]}
                   onPress={handleCancel}
                 >
                   <Text style={[styles.buttonText, styles.buttonTextSecondary]}>
                     {config.cancelText}
                   </Text>
                 </Pressable>
                 <Pressable
                   style={[styles.button, styles.buttonPrimary, styles.buttonHalf]}
                   onPress={handleConfirm}
                 >
                   <Text style={[styles.buttonText, styles.buttonTextPrimary]}>
                     {config.confirmText}
                   </Text>
                 </Pressable>
               </View>
             )}
           </View>
         </View>
       </Modal>
     );
   }
   ```

사용 예시 (실제 화면에서 모달 연결)

```typescript
// components/timecapsule-create/components/step-info/index.tsx
import React, { useState } from 'react';
import ConfirmModal from '@/components/timecapsule-create/components/confirm-modal';

function StepInfo() {
  // 모달 상태 관리
  const [showPaymentComplete, setShowPaymentComplete] = useState(false);
  const [showSubmitConfirm, setShowSubmitConfirm] = useState(false);
  const [showSubmitComplete, setShowSubmitComplete] = useState(false);

  // 결제 처리
  const handlePayment = async () => {
    try {
      // 결제 API 호출
      await paymentAPI();
      // 결제 성공 → PAYMENT_COMPLETE 모달 표시
      setShowPaymentComplete(true);
    } catch (error) {
      console.error('결제 실패:', error);
    }
  };

  // 제출 버튼 클릭
  const handleSubmitClick = () => {
    // SUBMIT_CONFIRM 모달 표시 (2개 버튼)
    setShowSubmitConfirm(true);
  };

  // 제출 확인 후 실제 제출 처리
  const handleSubmit = async () => {
    // 모달 먼저 닫기
    setShowSubmitConfirm(false);

    try {
      // 제출 API 호출
      await submitTimeCapsule();
      // 제출 성공 → SUBMIT_COMPLETE 모달 표시
      setShowSubmitComplete(true);
    } catch (error) {
      console.error('제출 실패:', error);
    }
  };

  return (
    <View>
      {/* 결제하기 버튼 */}
      <Pressable onPress={handlePayment}>
        <Text>결제하기</Text>
      </Pressable>

      {/* 제출하기 버튼 */}
      <Pressable onPress={handleSubmitClick}>
        <Text>제출하기</Text>
      </Pressable>

      {/* ==================== 모달들 ==================== */}

      {/* 1. 결제완료 확인 모달 (버튼 1개) */}
      <ConfirmModal
        visible={showPaymentComplete}
        type="PAYMENT_COMPLETE"
        onConfirm={() => {
          setShowPaymentComplete(false);
          // 다음 화면으로 이동 (예: navigation.navigate('NextStep'))
        }}
        onClose={() => setShowPaymentComplete(false)}
      />

      {/* 2. 제출 전 재확인 모달 (버튼 2개) */}
      <ConfirmModal
        visible={showSubmitConfirm}
        type="SUBMIT_CONFIRM"
        onConfirm={handleSubmit} // 제출 버튼 클릭 시 실제 제출 처리
        onCancel={() => setShowSubmitConfirm(false)} // 취소 버튼 클릭 시 모달만 닫기
        // closeOnBackdropPress: false이므로 onClose 생략 가능
      />

      {/* 3. 제출완료 확인 모달 (버튼 1개) */}
      <ConfirmModal
        visible={showSubmitComplete}
        type="SUBMIT_COMPLETE"
        onConfirm={() => {
          setShowSubmitComplete(false);
          // 홈 화면 또는 목록 화면으로 이동 (예: navigation.navigate('Home'))
        }}
        onClose={() => setShowSubmitComplete(false)}
      />
    </View>
  );
}
```
